## FOR TERMUX/UBUNTU/SSH USER
```bash
apt update && apt upgrade
apt install git -y
apt install nodejs -y
apt install ffmpeg -y
apt install imagemagick
apt install libwebp
git clone https://github.com/Fajar-ID/wa-bot2
cd wa-bot2
npm install
node index.js
```
